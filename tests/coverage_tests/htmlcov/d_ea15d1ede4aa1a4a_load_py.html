131 @staticmethod 

132 def check_file_structure(data: List) -> None: 

133 """ Checks if the file structure is valid. 

134 """ 

135 if not isinstance(data, list): 

136 raise InvalidFileEstucture("Invalid file structure") 

137 

138 for project in data: 

139 if not all(key in project for key in ['project', 'end_date', 'description', 'tasks']): 

140 raise InvalidFileEstucture("Invalid project format.") 

141 

142 if not isinstance(project['tasks'], list): 

143 raise InvalidFileEstucture("tasks must be a list.") 

144 

145 for task in project['tasks']: 

146 if not all(key in task for key in ['task', 'priority', 'end_date',\ 

147 'notification_date', 'description', 'subtasks']): 

148 raise InvalidFileEstucture("Invalid task format.") 

149 

150 if not isinstance(task['subtasks'], list): 

151 raise InvalidFileEstucture("subtasks must be a list.")

156 """ Checks if the name of some project to import already exists in the user's projects.
